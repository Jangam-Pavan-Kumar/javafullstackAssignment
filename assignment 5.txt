import java.util.Scanner;

        class Student {
	private static Scanner sc;
	public static void main(String[] args) 
	{
		int physics,chemistry,maths,biology; 
	    float total, Percentage;
		Scanner sc = new Scanner(System.in);
		
		System.out.print(" Please Enter the Four Subjects Marks : ");
		physics = sc.nextInt();	
		chemistry = sc.nextInt();	
		maths = sc.nextInt();	
		biology = sc.nextInt();	
		total = physics+chemistry+ maths+biology;
	    Percentage = (total / 4) * 100;
	 
	    System.out.println(" Total Marks =  " + total);
	    System.out.println(" Marks Percentage =  " + Percentage);
		
		if(Percentage >= 90)
	    {
			System.out.println("\n Grade A+");
		}
		else if(Percentage >= 80)
	    {
			System.out.println("\n Grade A");
		}
		else if(Percentage >= 70)
	    {
			System.out.println("\n Grade B");
		}
		else if(Percentage >= 55)
	    {
			System.out.println("\n Grade C");
		}
	
		else 
	    {
			System.out.println("\n Grade D");
		} 
	}
}
